CREATING DATABASE:
CREATE DATABASE <database_name>(
id int auto_increment,
name varchar(24),
primary key (id)
);


CREATING TABLE:
CREATE TABLE <database_name>(
id int auto_increment,
name varchar(24),
primary key (id)
);

ALTER TABLE <table_name>
